Sneaker-React-FE/
├── public/                       # Tài nguyên tĩnh (ảnh, favicon, fonts...)
├── src/                          # Thư mục mã nguồn chính
│   ├── api/                      # Cấu hình API và các endpoint gọi đến backend
│   ├── components/               # Các thành phần giao diện (UI Components)
│   │   ├── admin/                # Thành phần dành riêng cho quản trị (Sidebar, Navbar...)
│   │   ├── Common/               # Các thành phần dùng chung (LazyLoader, etc.)
│   │   ├── customer/             # Thành phần dành cho khách hàng
│   │   └── ui/                   # Các thành phần UI cơ bản (Shadcn UI components)
│   ├── context/                  # Quản lý trạng thái toàn cục bằng React Context
│   ├── hooks/                    # Các Custom Hooks xử lý logic và gọi API (TanStack Query hooks)
│   ├── interface/                # Định nghĩa các interface/type cho dữ liệu (TS interfaces)
│   ├── layouts/                  # Các khung giao diện chính (AdminLayout, RootLayout, StaffLayout)
│   ├── lib/                      # Cấu hình thư viện bên thứ ba và utilities (Axios, cn...)
│   ├── pages/                    # Các trang hiển thị chính của ứng dụng
│   │   ├── admin/                # Trang quản lý dành cho Admin (Statistics, Accounts, Products...)
│   │   ├── auth/                 # Các trang xác thực (Login, Register, Forget Password)
│   │   ├── home/                 # Trang chủ của người dùng
│   │   ├── products/             # Trang danh sách và chi tiết sản phẩm khách hàng
│   │   ├── orders/               # Quản lý đơn hàng khách hàng
│   │   └── returns/              # Quản lý đổi trả khách hàng
│   ├── routes/                   # Cấu hình định tuyến (React Router DOM)
│   ├── services/                 # Xử lý các dịch vụ phụ trợ
│   ├── stores/                   # Quản lý trạng thái bằng Zustand (Cart, Pending Carts...)
│   ├── utils/                    # Các hàm tiện ích dùng chung (Format currency, date...)
│   ├── App.tsx                   # Thành phần gốc của ứng dụng
│   └── main.tsx                  # Điểm khởi đầu của ứng dụng (React entry point)
│
├── .env                          # Biến môi trường (VITE_API_URL, etc.)
├── package.json                  # Quản lý dependencies và scripts (npm run dev, build)
├── tailwind.config.ts            # Cấu hình Tailwind CSS cho giao diện
├── tsconfig.json                 # Cấu hình TypeScript cho project
├── vite.config.ts                # Cấu hình công cụ build/dev Vite
└── README.md                     # Tài liệu hướng dẫn và giới thiệu project
